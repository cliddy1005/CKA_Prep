const [mode, setMode] = useState(“menu”);

const [revealed, setRevealed] = useState(false);
const [scores, setScores] = useState([]);
const [showHint, setShowHint] = useState(false);
const [touchStart, setTouchStart] = useState(null);
const [swipeDir, setSwipeDir] = useState(null);
const [animating, setAnimating] = useState(false);
const [streak, setStreak] = useState(0);
const [browseOpen, setBrowseOpen] = useState(null);

const filteredQ = category === “all” ? QUESTIONS : QUESTIONS.filter(q => q.cat === category);

function startQuiz() {
const shuffled = […filteredQ].sort(() => Math.random() - 0.5);
setQuestions(shuffled);
setCurrent(0); setRevealed(false); setScores([]); setShowHint(false); setStreak(0);
setMode(“quiz”);
}

function handleScore(result) {
if (animating || !revealed) return;
const newStreak = result === “got” ? streak + 1 : 0;
setStreak(newStreak);
setAnimating(true);
setSwipeDir(result === “got” ? “right” : “left”);
const newScores = […scores, result];
setTimeout(() => {
setScores(newScores);
if (current + 1 >= questions.length) { setMode(“results”); }
else { setCurrent(current + 1); setRevealed(false); setShowHint(false); }
setSwipeDir(null); setAnimating(false);
}, 300);
}

function onTouchStart(e) { setTouchStart(e.touches[0].clientX); }
function onTouchEnd(e) {
if (!revealed || touchStart === null) return;
const diff = e.changedTouches[0].clientX - touchStart;
if (Math.abs(diff) > 60) handleScore(diff > 0 ? “got” : “missed”);
setTouchStart(null);
}

}

return (
<div style={s.root} onTouchStart={onTouchStart} onTouchEnd={onTouchEnd}>
<div style={s.topBar}>
<button style={s.backBtn} onClick={()=>setMode(“menu”)}>← Menu</button>
<span style={s.topBarTitle}>{CATEGORIES[category]}</span>
<span style={s.topBarCount}>{current+1}/{questions.length}</span>
</div>
<div style={s.progressTrack}><div style={{…s.progressFill,width:`${progress}%`}}/></div>
{streak>=3&&<div style={s.streak}>🔥 {streak} streak</div>}
<div style={{flex:1,display:“flex”,flexDirection:“column”,padding:“12px 20px”,gap:12}}>
<div style={{…s.card,…(swipeDir===“right”?s.cardRight:swipeDir===“left”?s.cardLeft:{})}} onClick={()=>!revealed&&setRevealed(true)}>
<div style={s.cardCat}>{CATEGORIES[q?.cat]}</div>
<div style={s.cardQ}>{q?.q}</div>
{!revealed&&<div style={s.tapHint}>tap to reveal answer</div>}
{revealed&&<><div style={s.divider}/><div style={s.cardA}>{q?.a}</div></>}
</div>
{!revealed&&(
<button style={s.hintBtn} onClick={e=>{e.stopPropagation();setShowHint(!showHint);}}>
{showHint?“Hide hint”:“💡 Hint”}
</button>
)}
{showHint&&!revealed&&<div style={s.hintBox}>{q?.hint}</div>}
{revealed&&(
<div style={s.scoreRow}>
<button style={s.missedBtn} onClick={()=>handleScore(“missed”)}><span style={{fontSize:22}}>✕</span><span>Missed</span></button>
<button style={s.gotBtn} onClick={()=>handleScore(“got”)}><span style={{fontSize:22}}>✓</span><span>Got it</span></button>
</div>
)}
<div style={s.dots}>
{questions.map((_,i)=>(
<div key={i} style={{…s.dot,…(i<scores.length?(scores[i]===“got”?s.dotGot:s.dotMissed):(i===current?s.dotCurrent:{}))})}/>
))}
</div>
</div>
</div>
);
}

ReactDOM.createRoot(document.getElementById(“root”)).render(<App/>);
